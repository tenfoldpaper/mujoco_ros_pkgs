cmake_minimum_required(VERSION 3.8)
project(mujoco_ros2_control)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(ament_cmake_ros REQUIRED)
find_package(mujoco_ros REQUIRED)
find_package(hardware_interface REQUIRED)
find_package(controller_manager REQUIRED)
find_package(mujoco 3.2.0 REQUIRED) # TODO: How to get away without explicitly linking mujoco here?
find_package(yaml-cpp REQUIRED)

find_package(rclcpp REQUIRED)
find_package(pluginlib REQUIRED)

pluginlib_export_plugin_description_file(mujoco_ros plugins.xml)

add_library(mujoco_ros2_control SHARED src/mujoco_ros2_control.cpp)
target_compile_features(mujoco_ros2_control PUBLIC c_std_99 cxx_std_17)  # Require C99 and C++17
target_include_directories(mujoco_ros2_control PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>
  ${YAML_CPP_INCLUDE_DIRS}
  /home/bimanual/leins_ws/build/mujoco_ros/include # so this solves not found error wrt mujoco_ros/ros_version.hpp
)

target_link_libraries(mujoco_ros2_control
  mujoco::mujoco
  ${YAML_CPP_LIBRARIES}
)

ament_target_dependencies(
  mujoco_ros2_control
  rclcpp
  hardware_interface
  controller_manager
  "mujoco_ros"
  "pluginlib"
)

install(
  DIRECTORY include/
  DESTINATION include
)
install(
  TARGETS mujoco_ros2_control
  # EXPORT export_${PROJECT_NAME}
  ARCHIVE DESTINATION lib
  LIBRARY DESTINATION lib
  RUNTIME DESTINATION bin
)
